A empresa cliente deverá possuir:
\begin{itemize}
\item{\hardware:}
\newline
  \hspace{2cm}- impressoras;
\newline
  \hspace{2cm}- acessórios (quando necessários);

\item{\software:}
\newline
  \hspace{2cm}- \drivers de impressão para os equipamentos.

\item{insumos para impressão:}
\newline
  \hspace{2cm}- papel;
\newline
  \hspace{2cm}- cartucho de tinta;
\newline
  \hspace{2cm}- \toner;
\newline
  \hspace{2cm}- \kits de manutenção;

\item{infraestrutura:}
\newline
  \hspace{2cm}- local seguro e adequado para armazenamento dos insumos de impressão;
\newline
  \hspace{2cm}- local adequado para instalação e manuseio das impressoras.
\end{itemize}
